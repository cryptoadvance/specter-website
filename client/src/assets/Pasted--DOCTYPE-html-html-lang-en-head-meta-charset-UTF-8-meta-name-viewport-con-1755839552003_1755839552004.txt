    <script>
        // Get references to all filter and vendor elements
        const continentFilter = document.getElementById('continent-filter');
        const tagsFilter = document.getElementById('tags-filter');
        const vendorCards = document.querySelectorAll('.vendor-card');

        // Function to filter vendors based on selected criteria
        function filterVendors() {
            const selectedContinent = continentFilter.value.toLowerCase();
            const selectedTag = tagsFilter.value.toLowerCase();

            vendorCards.forEach(card => {
                const cardContinent = card.dataset.continent.toLowerCase();
                const cardTags = card.dataset.tags.toLowerCase().split(' ');
                
                // Check if the card matches the continent filter
                const continentMatch = selectedContinent === 'all' || cardContinent === selectedContinent;

                // Check if the card matches the tags filter
                const tagMatch = selectedTag === 'all' || cardTags.includes(selectedTag);

                // Show the card if both filters match, otherwise hide it
                if (continentMatch && tagMatch) {
                    card.classList.remove('hidden');
                } else {
                    card.classList.add('hidden');
                }
            });
        }

        // Add event listeners to the filters to trigger the filtering function
        continentFilter.addEventListener('change', filterVendors);
        tagsFilter.addEventListener('change', filterVendors);

        // Initial call to set up the page state on load
        window.addEventListener('load', filterVendors);
    </script>
