    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const filterButtons = document.querySelectorAll('.filter-btn');
            const partsList = document.getElementById('parts-list');
            const parts = partsList.querySelectorAll('li[data-cases]');
            const allLegendSections = partsList.querySelectorAll('div[data-cases]');
            const caseSections = document.querySelectorAll('.case-section');
            const legendSection = document.getElementById('legend-section');
            const partLinks = document.querySelectorAll('.case-section ul a');

            // Set initial state to 'Show All'
            filterElements('all');

            // Function to handle the filtering logic
            function filterElements(filter) {
                // Synchronize the active state of all filter buttons
                filterButtons.forEach(btn => {
                    if (btn.dataset.filter === filter) {
                        btn.classList.add('active');
                    } else {
                        btn.classList.remove('active');
                    }
                });

                // Handle the case sections
                caseSections.forEach(section => {
                    const caseType = section.dataset.case;
                    if (filter === 'all' || caseType === filter || (filter === 'specter-shield-lite-battery' && caseType === 'specter-shield-lite')) {
                        section.style.display = 'block';
                    } else {
                        section.style.display = 'none';
                    }
                });

                // Handle the legend of parts visibility
                legendSection.style.display = 'block';

                // Filter individual parts within the legend
                parts.forEach(part => {
                    const requiredCases = part.dataset.cases.split(' ');
                    if (filter === 'all' || requiredCases.includes(filter) || (filter === 'specter-shield-lite-battery' && requiredCases.includes('specter-shield-lite'))) {
                        part.style.display = 'block';
                    } else {
                        part.style.display = 'none';
                    }
                });

                // Filter case list entries within the legend
                allLegendSections.forEach(section => {
                    const requiredCases = section.dataset.cases.split(' ');
                    if (filter === 'all' || requiredCases.includes(filter) || (filter === 'specter-shield-lite-battery' && requiredCases.includes('specter-shield-lite'))) {
                        section.style.display = 'block';
                    } else {
                        section.style.display = 'none';
                    }
                });
            }

            // Event listener for filter buttons
            filterButtons.forEach(button => {
                button.addEventListener('click', () => {
                    const filter = button.dataset.filter;
                    filterElements(filter);
                });
            });

            // Function to highlight the target element
            function highlightTarget(targetId) {
                const targetElement = document.getElementById(targetId);
                if (targetElement) {
                    // Add the highlight class
                    targetElement.classList.add('highlight');
                    // Remove the highlight class after a short delay
                    setTimeout(() => {
                        targetElement.classList.remove('highlight');
                    }, 2000); // Highlight for 2 seconds
                }
            }

            // Event listener for part links inside the case sections
            partLinks.forEach(link => {
                link.addEventListener('click', (event) => {
                    // Prevent default anchor link behavior
                    event.preventDefault();

                    // Get the target ID from the href
                    const targetId = link.getAttribute('href').substring(1);

                    // Set filter to 'all' to ensure the target is visible
                    filterElements('all');

                    // Scroll to the element and highlight it
                    document.getElementById(targetId).scrollIntoView({
                        behavior: 'smooth'
                    });
                    
                    // Delay the highlight slightly to ensure scroll has started
                    setTimeout(() => {
                        highlightTarget(targetId);
                    }, 300);
                });
            });
        });
